# Question 13

**Write a Java program to create a GUI application with:
a)​ A JComboBox containing several course names
b)​ A button labeled "Show Selection"
c)​ A label**

---

### Solution

// TODO: Write solution here.

```java
import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

public class ComboBoxApp extends JFrame {
    JComboBox<String> courseComboBox;
    JButton btnShowSelection;
    JLabel lblSelection;

    public ComboBoxApp() {
        setTitle("Course Selection App");
        setSize(350, 150);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setLayout(new FlowLayout());

        // a) A JComboBox containing several course names
        String[] courses = {"Core Java", "Advanced Java", "Spring Boot", "Hibernate", "Python"};
        courseComboBox = new JComboBox<>(courses);
        
        // b) A button labeled "Show Selection"
        btnShowSelection = new JButton("Show Selection");
        
        // c) A label to show result
        lblSelection = new JLabel("Please select a course");

        // Add components
        add(new JLabel("Select Course: "));
        add(courseComboBox);
        add(btnShowSelection);
        add(lblSelection);

        // Event Handling
        btnShowSelection.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                String selectedCourse = (String) courseComboBox.getSelectedItem();
                lblSelection.setText("You selected: " + selectedCourse);
            }
        });
    }

    public static void main(String[] args) {
        new ComboBoxApp().setVisible(true);
    }
}
```
